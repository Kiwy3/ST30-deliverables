\begin{frame}{Strategie de Recherche : Optimisation Bayésienne par Process Gaussien }
    \begin{columns}
    %%%%%%%%%%%%%%%%%%%%%%%%%% COLONNE DE GAUCHE %%%%%%%%%%%%%%
    \begin{column}[t]{0.45\textwidth} 
        \vspace*{5pt}
        \begin{block}{Principe}
            Utiliser un substitut moins cher à optimiser pour explorer l'espace de recherche
        \end{block}
        \hfill
        \begin{block}{\onslide<1->{Algorithme}}
                \begin{itemize}
                    \item<1-> Echantillon de $n$ Points (LHS)
                    \begin{itemize}
                        \item<2-> Entrainer le Process Gaussien (GP)
                        \item<3-> Optimiser la fonction d'acquisition
                        \item<3-> Evaluer ce nouveaux point
                    \end{itemize}
                \end{itemize}
        \end{block}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% Search Strategy : BaMSOO %%%%%%%%%%%%%%%%%%
\begin{frame}{Stratégie de Recherche : Bayesian Multi-Scale Optimistic Optimization (BaMSOO)}
    \begin{columns}
        %%%%%%%%%%%%%%%%%%%%%%%%%% COLONNE DE GAUCHE %%%%%%%%%%%%%%
        
        \begin{column}{0.45\textwidth} 
            \vspace*{10pt}

            Décomposition suivant SOO, mais utilisant des Process Gaussien pour éviter les évaluations non prometteuses. 
            \begin{block}{Evaluation par BaMSOO}
                \begin{itemize}
                    \item If $UCB(x) > f^+$ :
                    \begin{itemize}
                        \item $g(x) = f(x)$ {\small \textit real evaluation}
                    \end{itemize}
                    \item Else : 
                    \begin{itemize}
                        \item $g(x) = LCB(x)$ {\small \textit use LCB to replace $f(x)$}
                    \end{itemize}
                \end{itemize}
                
            \end{block}
     
            \end{column}
